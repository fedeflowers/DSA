The query breaks down into several parts using Common Table Expressions (CTEs):

  - The first CTE (`r`) has time complexity of O(N) for scanning the table once.

# leaf
with r as (
    select id, "Root" as type
    from tree
    where p_id is null
),
i as (
    select t1.id, "Inner" as type
    from tree t1 left join tree t2
    on t1.id = t2.p_id
    where t2.id is not null and t1.p_id is not null
),
l as (
    select t1.id, "Leaf" as type
    from tree t1 left join tree t2
    on t1.id = t2.p_id
    where t2.id is null
    except
    select id, "Leaf"
    from r
)

SELECT * FROM r
UNION
SELECT * FROM i
UNION
SELECT * FROM l
